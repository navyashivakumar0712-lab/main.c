// 2D Character Graphics Editor
#include <stdio.h>
#include <stdlib.h>

void draw_circle(int cx, int cy, int r) {
  int x = r, y = 0;
  int err = 1 - r;
  while (x >= y) {
    plot(cx+x, cy+y, '*');
    plot(cx-x, cy+y, '*');
    // Bresenham Midpoint Circle
    y++;
    if (err <= 0)  err += 2*y+1;
    else         { x--; err += 2*(y-x)+1; }
  }
}
